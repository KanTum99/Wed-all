Quick Deploy

One Command. Full Stack.

// ติดตั้ง MiMo Agent Teams
$ npm install @xiaomi/mimo-agent-teams@2.6
// หรือ ใช้ yarn
$ yarn add @xiaomi/mimo-agent-teams@2.6
// ตรวจสอบ version
$ mimo --version
→ v2.6.0 (1T params · 1M context)
// mimo.config.ts — กำหนด Agent Team
import { AgentTeam, Role } from '@xiaomi/mimo-agent-teams';
export default new AgentTeam({
name: 'mimo-production-team',
engine: 'mimo-v2-pro',
orchestrator: 'consensus',
agents: [
{ role: Role.CodeArchitect, count: 3 },
{ role: Role.SecurityAuditor, count: 2 },
{ role: Role.DataGuardian, count: 4 },
{ role: Role.PerfEngineer, count: 2 },
],
monitor: {
reflective: true,
realtime: true,
logLevel: 'verbose',
}
});
// เริ่ม Agent Team
$ mimo deploy --team production --monitor
